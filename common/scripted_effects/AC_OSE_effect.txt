# Party popularity loss effects
OSE_unity_party_pop_loss_effect = {
	# Set party loss to a variable declared in the event/decision/focus triggering the effect, which should be an integer (0-100) and not 0.10 for example
	set_variable = { osea_party_pop_loss = osea_party_pop_loss_var }
	if = { # Liberty goes first since Liberty and Unity are both democratic, also is guaranteed at least a 2% increase
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_liberty_party_pop_gain
			min = 2 # 2% increase guaranteed
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_liberty_party_pop_gain }
	}
	if = { # Independents Second
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_independent_party_pop_gain
			min = 1
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_independent_party_pop_gain }
	}
	if = { # OBP Third
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_obp_party_pop_gain
			min = 1
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_obp_party_pop_gain }
	}
	if = { # OPP Fourth
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_opp_party_pop_gain
			min = 1
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_opp_party_pop_gain }
	}
	if = { # Monarchists Fifth
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_monarchist_party_pop_gain
			min = 1
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_monarchist_party_pop_gain }
	}
	if = { # Socialists Last
		limit = {
			check_variable = { osea_party_pop_loss > 0 }
		}
		set_variable_to_random = {
			var = osea_socialist_party_pop_gain
			min = 1
			max = osea_party_pop_loss_var
			integer = yes
		}
		subtract_from_variable = { osea_party_pop_loss = osea_socialist_party_pop_gain }
	}

	# Divide gains by 100 so they work with the party popularity system
	divide_variable = { osea_unity_party_pop_gain = 100 }
	divide_variable = { osea_socialist_party_pop_gain = 100 }
	divide_variable = { osea_independent_party_pop_gain = 100 }
	divide_variable = { osea_obp_party_pop_gain = 100 }
	divide_variable = { osea_opp_party_pop_gain = 100 }
	divide_variable = { osea_monarchist_party_pop_gain = 100 }
	# Divide original party pop loss variable by 100
	divide_variable = { osea_party_pop_loss_var = 100 }

	## Add/Subtract popularities from all parties
	# Unity is the subject of the effect, so Unity loses populartity
	subtract_from_variable = { osea_unity_party_popularity_democratic = osea_party_pop_loss_var }
	# All other parties gain popularity
	add_to_variable = { osea_liberty_party_popularity_democratic = osea_liberty_party_pop_gain }
	add_to_variable = { osea_obp_party_popularity_fascism = osea_obp_party_pop_gain }
	add_to_variable = { osea_opp_party_popularity_fascism = osea_opp_party_pop_gain }
	add_to_variable = { osea_socialist_party_popularity = osea_socialist_party_pop_gain }
	add_to_variable = { osea_monarchist_party_popularity = osea_monarchist_party_pop_gain }
	add_to_variable = { osea_independent_popularity = osea_independent_party_pop_gain }

	# Clamp all variables so they don't go over 1 (100% popularity) or below 0 (0%) popularity
	clamp_variable = { # Unity
		var = osea_unity_party_popularity_democratic
