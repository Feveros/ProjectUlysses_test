defined_text = {
	name = GetMobilizationDivisionCount
	text = {
		trigger = {
            if = {
                limit = {
                    state_population_k < 999
                }
                set_temp_variable = { num_units = 1 }
            }
            else = {
                set_temp_variable = { num_units = state_population_k }
                multiply_temp_variable = { num_units = 0.01 }
                divide_temp_variable = { num_units = 9 }
                round_temp_variable = num_units
            }
		}
		localization_key = "[?num_units|G0]"
	}
}
