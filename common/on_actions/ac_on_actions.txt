on_actions = {
	on_startup = {
		effect = {
			### USTIO RESURCES - EVENT CHAIN FOR BELKAN WAR
			PLH = {
				country_event = { id = trigger_event.1 days = 1480 }
			}
			every_possible_country = {
				set_rule = {
					can_join_factions = no
				}
			}
			BLK = {
				add_opinion_modifier = { target = OSE modifier = historical_enemies }
				}
			OSE = {
				add_opinion_modifier = { target = BLK modifier = historical_enemies }
			}
			BLK = {
				add_opinion_modifier = { target = PBL modifier = historical_friends }
				}
			PBL = {
				add_opinion_modifier = { target = BLK modifier = historical_friends }
			}
		}
	}
	on_capitulation = {
		effect = {
			if = {
				limit = {
					is_in_faction = yes
					all_allied_country = {
						NOT = {
							has_war_with = FROM
						}
					}
				}
				leave_faction = yes
			}
		}
	}
	on_daily_REC = {
		effect = {
			if = { #Recta peace with belka
				limit = {
					has_war_with = BLK
					has_global_flag = belkan_war_is_happeningz
					surrender_progress > 0.5
				}
				BLK = {
					transfer_state = 293
					transfer_state = 316
				}
				white_peace = BLK
				diplomatic_relation = {
					country = GEB
					relation = military_access
					active = no
				}
				diplomatic_relation = {
					country = UST
					relation = military_access
					active = no
				}
			}
		}
	}
	on_daily_FAT = {
		effect = {
			if = { #FATO peace with belka
				limit = {
					has_war_with = BLK
					has_global_flag = belkan_war_is_happening
					surrender_progress > 0.8
				}
				BLK = {
					transfer_state = 220
				}
				white_peace = BLK
				diplomatic_relation = {
					country = GEB
					relation = military_access
					active = no
				}
			}
		}
	}
	on_daily_GEB = {
		effect = {
			if = { #Gebet peace with belka
				limit = {
					has_war_with = BLK
					has_global_flag = belkan_war_is_happening
					surrender_progress > 0.5
				}
				BLK = {
					transfer_state = 256
				}
				white_peace = BLK
				diplomatic_relation = {
					country = FAT
					relation = military_access
					active = no
				}
				diplomatic_relation = {
					country = REC
					relation = military_access
					active = no
				}
			}
		}
	}
}
