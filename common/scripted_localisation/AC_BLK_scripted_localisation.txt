defined_text = {
	name = GetBLKExcaliburStatus
	text = {
		trigger = {
			has_global_flag = excalibur_destroyed
		}
		localization_key = excalibur_status_destroyed
	}
	text = {
		trigger = {
			OR = {
				has_active_mission = BLK_excalibur_cooling_after_aa
				has_active_mission = BLK_excalibur_cooling_after_bombing
			}
		}
		localization_key = excalibur_status_cooling
	}
	text = {
		trigger = {
			OR = {
				has_active_mission = BLK_excalibur_fire_during_aa
			}
		}
		localization_key = excalibur_status_fire
	}
	text = {
		trigger = {
			AND = {
				NOT = { has_active_mission = BLK_excalibur_fire_during_aa }
				NOT = { has_active_mission = BLK_excalibur_cooling_after_bombing }
				NOT = { has_active_mission = BLK_excalibur_cooling_after_aa }
				NOT = { has_global_flag = excalibur_destroyed }
			}
		}
		localization_key = excalibur_status_ready_fire
	}
}
