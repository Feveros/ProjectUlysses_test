    GEB_mobilize_reserves = {
        icon = placeholder 
        state_target = yes
        target_root_trigger = {
			has_completed_focus = GEB_reinstate_conscription
		}
		target_trigger = {
			FROM = {
				is_controlled_by = ROOT
                is_core_of = ROOT
				NOT = { has_state_flag = state_infantry_mobilized }
			}
		}
        available = {
            OR = {
                has_defensive_war = yes
                any_other_country = {
                    OR = {
                        is_justifying_wargoal_against = ROOT
                        has_wargoal_against = ROOT
                    }
                }
            }
        }
        on_map_mode = map_only
        days_remove = 14
        remove_effect = {
            FROM = {
                if = {
                    limit = {
                        state_population_k < 900

                    create_unit = {
                        division = "name = \"Conscript Division\" division_template = \"Conscript Division\" start_experience_factor = 0.2"
                        allow_spawning_on_enemy_provs = no
                        owner = ROOT
                        id = draft_infantry_id
                    }
                }
                else = {
                    set_variable = { recruitable_pop = state_population_k }
                    multiply_variable = { recruitable_pop = 0.01 }
                    set_variable = { units_to_spawn = recruitable_pop }
                    divide_variable = { units_to_spawn = 9 }
                    for_loop_effect = {
                        start = 1
                        end = units_to_spawn
                        compare = less_than_or_equals
                        create_unit = {
                            division = "name = \"Conscript Division\" division_template = \"Conscript Division\" start_experience_factor = 0.2"
                            allow_spawning_on_enemy_provs = no
                            owner = ROOT
                            id = draft_infantry_id
                        }
                    }
                }
                set_state_flag = state_infantry_mobilized
            }
        }
    }

GEB_land_reform = {
    GEB_land_reform_in_state = {
        icon = placeholder
        cost = 100
        state_target = yes
        target_root_trigger = {
			has_completed_focus = GEB_continue_land_reform
		}
		target_trigger = {
			FROM = {
				is_controlled_by = ROOT
				OR = {
                    has_dynamic_modifier = { modifier = geb_aristocrat_held_land_2 }
                    has_dynamic_modifier = { modifier = geb_aristocrat_held_land_1 }
                }
			}
		}
        days_remove = var:land_reform_time
        modifier = {
            political_power_factor = -0.10
        }
        ai_will_do = {
            base = 100
        }
        remove_effect = {
            FROM = {
                add_extra_state_shared_building_slots = 1
                if = {
                    limit = {
                        FROM = {
                            has_dynamic_modifier = { modifier = geb_aristocrat_held_land_2 }
                        }
                    }
                    remove_dynamic_modifier = { modifier = geb_aristocrat_held_land_2 }
                }
                else = {
                    remove_dynamic_modifier = { modifier = geb_aristocrat_held_land_1 }
                }
            }
            add_to_variable = { land_reform_points = 1 }
            if = {
                limit = {
                    land_reform_points > 2
                }
                set_country_flag = GEB_land_reform_complete
            }
        }
    }
}
