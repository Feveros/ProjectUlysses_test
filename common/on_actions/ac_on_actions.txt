on_actions = {
	on_startup = {
		effect = {
			### USTIO RESURCES - EVENT CHAIN FOR BELKAN WAR
			PLH = {
				country_event = { id = trigger_event.1 days = 1480 }
			}
			every_possible_country = {
				set_rule = {
					can_join_factions = no
				}
			}
			BLK = {
				add_opinion_modifier = { target = OSE modifier = historical_enemies }
				}
			OSE = {
				add_opinion_modifier = { target = BLK modifier = historical_enemies }
			}
			BLK = {
				add_opinion_modifier = { target = PBL modifier = historical_friends }
				}
			PBL = {
				add_opinion_modifier = { target = BLK modifier = historical_friends }
			}
		}
	}
	on_capitulation = {
		effect = {
			if = {
				limit = {
					is_in_faction = yes
					all_allied_country = {
						NOT = {
							has_war_with = FROM
						}
					}
				}
				leave_faction = yes
			}
			if = { #Recta peace with belka
				limit = {
					ROOT = {
						original_tag = REC
						has_war_with = BLK
						has_global_flag = belkan_war_is_happening
					}
				}
				BLK = {
					transfer_state = 293
					transfer_state = 316
				}
				ROOT = {
					white_peace = BLK
					diplomatic_relation = {
						country = GEB
						relation = military_access
						active = no
					}
					diplomatic_relation = {
						country = UST
						relation = military_access
						active = no
					}
				}
			}
			else_if = { #FATO peace with belka
				limit = {
					ROOT = {
						original_tag = FAT
						has_war_with = BLK
						has_global_flag = belkan_war_is_happening
					}
				}
				BLK = {
					transfer_state = 220
				}
				ROOT = {
					white_peace = BLK
					diplomatic_relation = {
						country = GEB
						relation = military_access
						active = no
					}
				}
			}
			else_if = { #Gebet peace with belka
				limit = {
					ROOT = {
						original_tag = GEB
						has_war_with = BLK
						has_global_flag = belkan_war_is_happening
					}
				}
				BLK = {
					transfer_state = 256
				}
				ROOT = { 
					white_peace = BLK
					diplomatic_relation = {
						country = FAT
						relation = military_access
						active = no
					}
					diplomatic_relation = {
						country = REC
						relation = military_access
						active = no
					}
				}
			}
		}
	}
}
